Getting the Hang of Linked Lists: A Code Walkthrough

Singly Linked List (single.cpp)

The Node Class

This is the basic building block. Each node holds a piece of data and a pointer to the next node in the chain. It's the fundamental unit of the list.

class Node{
    public:
        int data;
        Node* next;

        Node(int data1){
            data = data1;
            next = nullptr;
        }
        Node(int data1, Node* next1){
            data = data1;
            next = next1;
        }
};

convrertArr2LL

Turns a vector (like a dynamic array) into a brand new linked list. This is a common way to initialize a list with some starting data.

Time Complexity: O(n) - It has to loop through every element in the input array once.

Node* convrertArr2LL( vector<int> &arr){
    Node* head = new Node(arr[0]);
    Node* mover = head;
    for(int i = 1;i<arr.size(); i++){
        Node* temp = new Node(arr[i]);
        mover->next = temp;
        mover = temp;
    }
    return head;
}

removeshead

Removes the first node (the head) of the list. This is a very fast operation.

Time Complexity: O(1) - It doesn't need to loop through the list; it just reassigns the head pointer.

Node* removeshead(Node*head){
    Node* temp = head;
    head = head-> next;
    delete temp;
    return head;
}

removeK

Removes the node at the k-th position in the list. To do this, it has to travel from the head to the desired position.

Time Complexity: O(k) - The time taken depends on how far down the list 'k' is. In the worst case (removing the last element), it's O(n).

Node* removeK(Node* head , int k){
    if(head == NULL) return nullptr;
    if(k==1){
        Node* temp = head;
        head = head->next;
        free(temp);
        return head;
    }
    int count = 0;
    Node* temp = head;
    Node* prev = NULL;
    while(temp != NULL){
        count ++;
        if (count == k){
            prev->next = prev->next->next;  //i.e prev->next->next = temp->next;
            free(temp);
            break;
        }
        prev = temp;
        temp = temp -> next;
    }
    return head;
}

remove_elemnt

Finds the first node with a specific value and removes it. This requires searching through the list.

Time Complexity: O(n) - In the worst-case scenario, it might have to check every single node to find the value.

Node* remove_elemnt(Node* head ,int k) {
    if(head->data == NULL) return nullptr;
    if(head->data == k){
        Node* temp = head;
        head = head->next;
        free(temp);
        return head;
    } 
    Node* temp = head;
    Node* prev = NULL;
    while(temp != NULL){ 
        if (temp-> data == k){
            prev->next = prev->next->next;  // prev->next->next = temp->next;
            free(temp);
            break;
        }
        prev = temp;
        temp = temp -> next;
    }
    return head;
}

removestail

Removes the very last node of the list. In a singly linked list, this is inefficient because you have to travel all the way to the second-to-last node to change its `next` pointer.

Time Complexity: O(n) - It has to traverse the entire list to find the end.

Node* removestail(Node* head){
    if(head == NULL || head->next == NULL)
        return NULL;
    Node* temp = head;
    while(temp->next->next != NULL){
        temp = temp ->next;
    }
    delete temp-> next;
    temp->next = nullptr;   
    return head;
}

insertAtFirst

Adds a new node to the very beginning of the list. This is one of the main advantages of a linked list, as it's extremely fast.

Time Complexity: O(1) - It only involves creating a new node and updating the head pointer, no looping required.

Node* insertAtFirst(Node* head , int val){
    Node* temp = new Node(val, head);
    return temp;
}

insertAtLast

Adds a new node to the very end of the list. Similar to removing the tail, this requires traversing the whole list first.

Time Complexity: O(n) - You have to get to the end of the list before you can add the new node.

Node* insertAtLast(Node* head, int val){
    if(head == NULL){
        return new Node(val);
    }
    Node* temp = head;
    while(temp->next != NULL){
        temp-> next;
    }
    Node* newNode = new Node(val);
    temp->next = newNode;
    return head;
}

insertAtK

Inserts a new node at the k-th position.

Time Complexity: O(k) - Requires traversing to the (k-1)th node.

Node* insertAtK(Node* head , int val , int k){
    if(head==NULL){
        return new Node(val);
    }
    if(k == 1){
        return new Node(val , head);
    }
    int count = 0;
    Node* temp = head;
    while(temp!= NULL){
        count++;
        if(count == k-1){
            Node* x = new Node(val , temp->next);
            temp-> next = x;
            break;
        }
        temp = temp-> next;
    }
    return head;
}

insertAtval

Tries to insert a new node before a node that has a specific value. (Note the bug: `if(temp->next->data = val)` should be `==` for comparison!)

Time Complexity: O(n) - May need to search the entire list.

Node* insertAtval(Node* head , int val){
    if(head==NULL){
        return NULL;
    }
    if(head->data==val){
        return new Node(val , head);
    }
    Node* temp = head;
    while(temp->next != NULL){
        if(temp->next->data = val){
            Node* x = new Node(val , temp->next);
            temp-> next = x;
            break;
        }
        temp = temp-> next;
    }
    return head;
}

Doubly Linked List (doubly.cpp)

The Node Class

This node is a bit more advanced. It has a pointer to the next node and also a pointer to the back (previous) node. This two-way link is what makes it "doubly" linked.

class Node{
    public:
        int data;
        Node* next;
        Node* back;

        Node(int data1){
            data = data1;
            next = nullptr;
            back = nullptr;
        }
        Node(int data1, Node* next1 , Node* back1){
            data = data1;
            next = next1;
            back = back1;
        }
};

convertArr2DLL

Turns a vector into a doubly linked list, making sure to link nodes in both directions.

Time Complexity: O(n) - Just like the singly linked version, it must iterate through the whole array.

Node* convertArr2DLL(vector <int> &arr){
    Node*head = new Node(arr[0]);
    Node* prev = head;
    for(int i = 1 ; i<arr.size(); i++){
        Node* temp = new Node(arr[i], nullptr , prev);
        prev ->next = temp;
        prev = temp;
    }
    return head;
}

deleteHead

Removes the first node and makes sure the new head's `back` pointer is null.

Time Complexity: O(1) - Very efficient, no traversal needed.

Node* deleteHead(Node* head){
    if(head== NULL || head->next ==NULL)
        return NULL;

    Node* prev = head;
    head = head->next;

    head->back = nullptr;
    prev-> next = nullptr;

    return(head);
    free(prev);
}

deletetail

Removes the last node of the list. This is much more efficient in a doubly linked list because you can instantly access the second-to-last node from the tail.

Time Complexity: O(n) - Your current implementation still traverses from the head. A more optimal version would keep a separate `tail` pointer, making this an O(1) operation.

Node* deletetail(Node* head){
    if(head== NULL || head->next ==NULL)
        return NULL;
    Node* tail = head;
    while(tail->next != NULL){
        tail = tail->next;
    }
    Node* newtail = tail->back;
    newtail->next = nullptr;
    tail->back = nullptr;
    delete tail;
    return head;
}

deleteKthElement

Removes the node at the k-th position. The logic is cleaner than the singly linked version because once you find the node, you can easily access its previous (`back`) and next (`front`) neighbors to re-link them.

Time Complexity: O(k) - Still requires traversing to the k-th element.

Node* deleteKthElement(Node* head , int k){
    if(head==NULL){
        return NULL;
    }
    int count = 0;
    Node* temp = head;
    while(temp != NULL){
        count ++;
        if(count == k) break;
        temp = temp ->next;
     }
    Node* prev = temp->back;
    Node* front = temp->next;

    if(prev == NULL && front == NULL){
        return NULL;
    }
    else if(prev == NULL){
        return deleteHead(head);
    }
    else if(front == NULL)
        return deletetail(head);
    
    prev->next = front;
    front->back = prev;

    temp->next = nullptr;
    temp->back = nullptr;
    free(temp);

    return (head);
}

deleteNode

A helper function to delete a specific node when you already have a pointer to it.

Time Complexity: O(1) - Extremely fast as you already know the node's location.

void deleteNode(Node* temp){
    Node* front = temp->next;
    Node* prev = temp->back;

    if(front == NULL){
        prev->next = nullptr;
        temp->back = nullptr;
        free(temp);
        return;
    }

    prev -> next = front;
    prev -> back = prev;

    temp->next = temp->back = nullptr;
    free(temp);
}

insertBeforeHead

Adds a new node to the front of the list.

Time Complexity: O(1) - A constant time operation.

Node* insertBeforeHead(Node* head , int val){
    Node* newNode = new Node(val , head , nullptr);
    head->back = newNode;

    return newNode;
}

insertBeforeTail

Inserts a new node right before the last node of the list.

Time Complexity: O(n) - Requires traversing to the end of the list first.

Node* insertBeforeTail(Node* head , int val){
    if(head-> next == NULL)
        return insertBeforeHead(head , val);
    Node* tail = head;
    while(tail-> next != NULL){
        tail = tail-> next ;
    }
    Node* prev = tail->back;
    Node* newNode = new Node(val, tail , prev);
    prev ->next = newNode;
    tail -> back = newNode;

    return head;
}

Node* insertBeforeK(Node* head , int k , int val ){

insertBeforeNode

Inserts a new node before a specific node that you provide a pointer to.

Time Complexity: O(1) - Very efficient as you don't need to search for the node.

void insertBeforeNode(Node* temp , int val){
    Node* prev = temp->back;
    Node* newNode = new Node(val , temp , prev);
    temp->back = newNode;
    prev->next = newNode;
}

print2DLL

Prints all the values in the doubly linked list from start to finish.

Time Complexity: O(n) - Must visit every node.

void print2DLL(Node* head){
    while(head != NULL){
        cout << head -> data << " ";
        head = head -> next;
    }
}
